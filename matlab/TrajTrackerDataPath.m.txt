function p = TrajTrackerDataPath
% TrajTrackerDataPath - define the base path for TrajTracker data files.
% 
% The paths of files/datasets in this toolbox are usually specified with
% respect to the base path, which is defined by this function.
% 
% If you have just installed the toolbox, this file is called
% "TrajTrackerDataPath.m.txt", so you should remove the ".txt" extension.

    p = '/path/to/your/base/data/directory';

end
